三、动态请求，封装myRequest方法，重写首页

1.在根目录下建立util/api.js

const BASE_URL = "http://localhost:8082"//域名
export const myRequest =(options) =>{
    return new Promise((resolve,reject)=>{
        uni.request({
            url:BASE_URL+options.url,
            method:options.method || 'GET',
            data: options.data || {},
            success:(res)=>{
                if(res.data.status !== 0){
                    return uni.showToast({
                        title:'获取数据失败'
                    })
                }
                resolve(res)
            },
            fail:(err)=>{
                return uni.showToast({
                    title:'请求接口失败'
                })
                reject(err)
            }
        })
    })
}
