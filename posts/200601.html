import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-systest',
  templateUrl: './systest.component.html',
  styleUrls: ['./systest.component.styl']
})
export class SystestComponent implements OnInit {
  //右下侧表格数据
  listOfData = [
    {
      name: '版本',
      status:0,
      router:'edition'
    },
    {
      name: '资源',
      status:1,
      router:'resources'
    },
    {
      name: '节点',
      status:0,
      router:'node'
    },
    {

      name: '权限',
      status:1,
      router:'jurisdiction'
    },
    {
      name: '配置',
      status:1,
      router:'config'
    },
    {
      name: '句柄',
      status:0,
      router:'handle'
    },
  ];
  constructor() { }

  ngOnInit() {
  }
  // 左侧系统环境检测条目下拉框默认展开状态
  openMap: { [name: string]: boolean } = {
    sub1: true,
    sub2: false,
    sub3: false
  };
  openHandler(value: string): void {
    for (const key in this.openMap) {
      if (key !== value) {
        this.openMap[key] = false;
      }
    }
  }
  // 左侧一键禁用按钮
  switchValue = false;
  isDisabled = true;
  //右侧一键开始检测-开始检测按钮
  isLoadingTwo = false;
  loadTwo(): void {
    this.isLoadingTwo = true;
    setTimeout(() => {
      this.isLoadingTwo = false;
    }, 5000);
  }
}
